# jOOQ Configuration Guide
# Add these properties to your application.properties for better jOOQ debugging and performance

# ===== jOOQ Settings =====

# Enable SQL query logging (DEBUG level shows the generated SQL)
logging.level.org.jooq.tools.LoggerListener=DEBUG

# Enable jOOQ feedback messages
logging.level.org.jooq=INFO

# ===== Usage Notes =====

# 1. jOOQ generates type-safe SQL at compile-time
# 2. Generated classes are in: src/main/java/com/quodex/_miles/jooq/generated/
# 3. Regenerate classes when database schema changes: mvn clean compile jooq:generate
# 4. Use DSLContext for all jOOQ queries
# 5. Import generated Tables for type-safe column references
#    Example: import static com.quodex._miles.jooq.generated.Tables.*;
